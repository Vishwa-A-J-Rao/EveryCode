#2 Two_To_Four_Decoder
module Decoder(y,i1,i0,e);
output [3:0]y;
input i1,i0;
input e;
wire i1not,i0not,eNot;
not G1(i1not,i1);
not G2(i0not,i0);
not G3(eNot,e);
nand G4(y[0],i1not,i0not,eNot);
nand G5(y[1],i1not,i0,eNot);
nand G6(y[2],i1,i0not,eNot);
nand G7(y[3],i1,i0,eNot);
endmodule
#4 8:3 Encoder
module encoder(y,x,e);
input e;
input [7:0]x;
output[2:0]y;
reg [2:0]y;
always @(e,x)
begin
if(e==1'b0)
y=3'b000;
else
begin
case(x)
8'b00000001:y=3'b000;
8'b00000010:y=3'b001;
8'b00000100:y=3'b010;
8'b00001000:y=3'b011;
8'b00010000:y=3'b100;
8'b00100000:y=3'b101;
8'b01000000:y=3'b110;
8'b10000000:y=3'b111;
default:y=3'bzzz;
endcase
end
end
endmodule
