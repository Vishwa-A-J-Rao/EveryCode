    3. HDL CODE TO DESIGN 32 BIT ALU
  • Verilog Code:
    • module alu(result,ack,a,b,opcode,en);
    input [2:0]opcode;
    input en;
    input [31:0]a,b;
    output [32:0]result;
    reg [32:0]result;
    output ack;
    reg ack;
    always @(en,opcode,a,b)
    begin
        if(en==0)
            result=33'bzzzz;
        else
        case(opcode)
            3'b000:result=a+b;
            3'b001:result=a-b;
            3'b010:result=a+1;
            3'b011:result=a-1;
            3'b100:result=a;
            3'b101:result=~a;
            3'b110:result=a|b;
            3'b111:result=a&b;
        endcase
        ack=(en==1)?1:0;
    end
endmodule

    8. WRITE HDL CODE TO CONTROL SPEED, DIRECTION OF DC MOTOR
  • Verilog Code:
    • module dcmotor(clk,reset,dir,row,rly,pwm);
    input clk,reset,dir;
    input [3:0] row;
    output rly;
    output [1:0]pwm;
    reg [1:0]pwm;
    reg[7:0] counter=8'b11111110;
    reg[16:0] div_reg;
    reg dclk;
    wire tick;
    integer duty_cycle;
    always@(posedge clk)
    begin
        div_reg = div_reg + 1;
        dclk=div_reg[12];
    end
    assign tick = row[0] & row[1] & row[2] & row[3];
    always@(negedge tick)
    begin
        case(row)
            4'b1110: duty_cycle = 255;
            4'b1101: duty_cycle = 200;
            4'b1011: duty_cycle = 150;
            4'b0111: duty_cycle = 100;
            default: duty_cycle = 100;
        endcase
    end
    always@(posedge dclk)
    begin
        if(reset==0)
            begin
                counter=8'b0;
                pwm=2'b01;
            end
        else
        begin
            counter = counter + 1;
            if (counter >= duty_cycle)
                pwm[1] = 0;
            else
                pwm[1] = 1;
        end
    end
    assign rly=dir;
endmodule

    • module stepper(clk,reset, dipsw1, dipsw2, dipsw3, dout);

    2. Linear and circular convolution of two given sequences, Commutative, distributive and associative property of convolution. 
    import numpy as np
from operator import mod
def comp_array(l,r):
    N=len(l)
    count=0
    for i in range(N):
        if(l[i]==r[i]):
            count+=1
        if(count==N):
            return True
        else:
            return False

def lc(x,h):
    m=len(x)
    n=len(h)
    N=m+n-1
    y=np.zeros((N))
    x=list(x)
    h=list(h)
    x=x+[0]*(N-m)
    h=h+[0]*(N-n)
    for n in range(N):
        for k in range(n+1):
            y[n]=y[n]+x[k]*h[n-k]
    return y

def circ_conv(x,h):
    N=len(x)
    y=np.zeros(N)
    for i in range (N):
        for j in range (N):
            y[i]=y[i]+x[j]*h[mod((i-j),N)]
    return y

def flip(x):
    N=len(x)
    y=np.zeros(N)
    for i in range (N):
        y[i]=x[N-1-i]
    return y
