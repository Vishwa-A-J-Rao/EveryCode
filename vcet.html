    2. Linear and circular convolution of two given sequences, Commutative, distributive and associative property of convolution. 
    import numpy as np
from operator import mod
def comp_array(l,r):
    N=len(l)
    count=0
    for i in range(N):
        if(l[i]==r[i]):
            count+=1
        if(count==N):
            return True
        else:
            return False

def lc(x,h):
    m=len(x)
    n=len(h)
    N=m+n-1
    y=np.zeros((N))
    x=list(x)
    h=list(h)
    x=x+[0]*(N-m)
    h=h+[0]*(N-n)
    for n in range(N):
        for k in range(n+1):
            y[n]=y[n]+x[k]*h[n-k]
    return y

def circ_conv(x,h):
    N=len(x)
    y=np.zeros(N)
    for i in range (N):
        for j in range (N):
            y[i]=y[i]+x[j]*h[mod((i-j),N)]
    return y

def flip(x):
    N=len(x)
    y=np.zeros(N)
    for i in range (N):
        y[i]=x[N-1-i]
    return y

a=[1,2,3]
b=[1,1,1]
c=[5,6,7]
a=np.array(a)
b=np.array(b)
c=np.array(c)
print("a=",a)
print("b=",b)
print("c=",c)
print()
print("Commutative Property:")
lhs=lc(a,b)
print("LHS=",lhs)
rhs=lc(b,a)
print("RHS=",rhs)
if comp_array(lhs,rhs):
    print("LHS=RHS, Commutative Property Is Satisfied")
else:
    print("Its not")
print()
print("Associative Property:")
lhs1=lc(lc(a,b),c)
print("LHS=",lhs1)
rhs1=lc(a,lc(b,c))
print("RHS=",rhs1)
if comp_array(lhs1,rhs1):
    print("LHS=RHS, Associative Property Is Satisfied")
else:
    print("Its not")
print()
print("Distributive Property:")
lhs2=lc(a,b+c)
print("LHS=",lhs2)
rhs2=lc(a,b)+lc(a,c)
print("RHS=",rhs2)
if comp_array(lhs2,rhs2):
    print("LHS=RHS, Distributive Property Is Satisfied")
else:
    print("Its not")
print()
x3=circ_conv(a,b)
print("Circular Convolution is:",x3)

    4. Solving a given difference equation. 
    import numpy as np
def filter(b,a,x):
    N=len(x)
    b=list(b)+[0]*(N-len(b))
    a=list(a)+[0]*(N-len(a))
    y=np.zeros((N))
    for n in range (N):
        term1=0
        term2=0
        for k in range (n+1):
            term1=term1-a[k]*y[n-k]
            term2=term2+b[k]*x[n-k]
        y[n]=term1+term2
    return y

a=np.array([1,-4,3])

x=np.array([1,2,3])
y=filter(b,a,x)
print("a=",a)
print("b=",b)
print("x=",x)
print("Differential Equation Y=",y)

Verification Of Sampling Theorem

Sampling theorem states that band limited signal f(t) can be reconstructed from its samples if the sampling frequency is greater than the two times the highest frequency component of the signal. i.e fs>=2fm
